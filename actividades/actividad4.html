# Implementaci√≥n del M√©todo del Punto Fijo en Python

def punto_fijo(g, r0, tol, max_iter):
    """
    Encuentra una ra√≠z por el m√©todo de punto fijo.
    
    Args:
        g (function): La funci√≥n de iteraci√≥n g(r).
        r0 (float): La aproximaci√≥n inicial.
        tol (float): La tolerancia para el criterio de parada.
        max_iter (int): El n√∫mero m√°ximo de iteraciones.
        
    Returns:
        float: La ra√≠z aproximada, o None si no converge.
    """
    print(f"{'n':<5}{'r_n':<15}{'r_n+1':<15}{'Error Absoluto':<20}")
    print("-" * 55)
    
    rn = r0
    for n in range(max_iter):
        rn_1 = g(rn)
        error_abs = abs(rn_1 - rn)
        
        print(f"{n:<5}{rn:<15.6f}{rn_1:<15.6f}{error_abs:<20.6f}")
        
        if error_abs < tol:
            print("\nConvergencia alcanzada.")
            return rn_1
            
        rn = rn_1
        
    print("\nEl m√©todo no convergi√≥ tras el m√°ximo de iteraciones.")
    return None

# --- Problema a resolver ---
# f(r) = (1+r)^12 - 2 = 0
# Se transforma a r = g(r)
# g(r) = 2 / (1+r)^11 - 1

g = lambda r: 2 / (1 + r)**11 - 1

# --- Par√°metros iniciales ---
[cite_start]r0 = 0.05  # Estimaci√≥n inicial del 5% [cite: 131]
tolerancia = 1e-6
iteraciones_max = 50

# --- Ejecuci√≥n del m√©todo ---
raiz = punto_fijo(g, r0, tolerancia, iteraciones_max)

if raiz is not None:
    print(f"\nLa ra√≠z aproximada es r ‚âà {raiz:.6f}")
    print(f"La tasa de inter√©s mensual requerida es de aproximadamente {raiz*100:.3f}%")
